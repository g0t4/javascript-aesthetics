Here's a sample implementation of a bubble sort algorithm in JavaScript: 

function comparator(a, b) {
    return a - b;
}

function bubSrt(arr, cmp) {
    cmp = cmp || comparator;
    var temp;
    for (var i = 0; i < arr.length; i += 1) {
        for (var j = i; j > 0; j -= 1) {
            if (cmp(arr[j], arr[j - 1]) < 0) {
                temp = arr[j];
                arr[j] = arr[j - 1];
                arr[j - 1] = temp;
            }
        }
    }
    return arr;
}

bubSrt([4, 6, 2, 10, 1]);
